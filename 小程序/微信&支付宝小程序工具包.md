# 支付宝小程序



## 一、微信工具包

#### 1.获取微信code

```js
/**
 * 获取微信code码
 * return {Promise Function} code
 * 	成功 {code:'', status: 200}
 * 	失败 {status: 400, err}
 */
getWxCode: () => {
	return new Promise((resolve, reject) => {
		wx.login({
			success(res) {
				resolve({
					status: 200,
					code: res.code
				});
			},
			fail(err) {
				resolve({
					status: 400,
					err: err
				});
			}
		})
	})
},

```

#### 2.获取微信个人用户信息

```js
/**
 * 获取目前授权状态信息 这里只判断状态码如果不是 200 就是没有拿到信息就好
 * 
 * 检测用户是否授权个人用户信息
 * 如果有授权返回中文的json格式个人信息
 * 如果无授权或者失败的几种情况
 * 	{status: 200, result: res.userInfo, msg: '获取用户信息成功'}
 * 	{status: 400, result: false, msg: '调取配置信息函数失败'}
 * 	{status: 400, result: false, msg: '没有授权'}
 * 	{status: 402, result: false, msg: '获取用户信息函数失败'}
 */
checkIsAuthUserInfo() {
  return new Promise((resolve, reject) => {
    wx.getSetting({
      success(res) {
        // console.log('成功');
        if (res.authSetting['scope.userInfo']) {
          // 授权信息
          wx.getUserInfo({
            lang: 'zh_CN',
            success(res) {
              // 授权了 并且调用成功
              resolve({ status: 200, result: res.userInfo, msg: '获取用户信息成功' });
            },
            fail(err) {
              // 授权了 但是调用方法失败了
              // console.log(err);
              resolve({ status: 402, result: false, msg: '获取用户信息函数失败' });
            }
          })
        } else {
          // 未授权
          resolve({ status: 400, result: false, msg: '没有授权' });
        }
      },
      fail(err) {
        // console.log("调取配置信息方法失败")
        resolve({ status: 400, result: false, msg: '调取配置信息函数失败' });
      }
    })
  })
}

```

#### 4.更新升级

```js
// 我的这些函数是在app.js 的 onshow中调用的

const updateManager = wx.getUpdateManager()
console.log('小程序更新');
console.log(updateManager);

updateManager.onCheckForUpdate(function(res) {
    // 请求完新版本信息的回调
    console.log(res.hasUpdate)
})

updateManager.onUpdateReady(function() {
    uni.showModal({
        title: '更新提示',
        content: '新版本已经准备好，是否重启应用？',
        success: function(res) {
            if (res.confirm) {
                // 新的版本已经下载好，调用 applyUpdate 应用新版本并重启
                updateManager.applyUpdate()
            }
        }
    })
})

updateManager.onUpdateFailed(function() {
    // 新的版本下载失败
    uni.showModal({
        title: '更新提示',
        content: '新版本下载失败',
        showCancel: false
    })
})
```





## 二、支付宝的工具包

#### 1.获取支付宝的code

```js
/**
* 获取支付宝code码
* return {Promise Function} code
* 	成功 {code:'', status: 200}
* 	失败 {status: 400, err}
*/
getWxCode: () => {
    return new Promise((resolve, reject) => {
        my.getAuthCode({
            scopes: 'auth_base', // auth_base  auth_user   auth_zhimaauth_zhima
            success(res) {
                console.log('获取成功的回调函数');
                console.log(res)
                resolve({
                    status: 200,
                    code: res.authCode
                });
            },
            fail(err) {
                console.log('获取失败的回调');
                console.log(err);
                resolve({
                    status: 400,
                    err: err
                });
            }
        })
    })
}
```

#### 2.获取支付宝授权状态信息

```js
/**
	 * 获取目前授权状态信息 这里只判断状态码如果不是 200 就是没有拿到信息就好
	 * 
	 * 检测用户是否授权个人用户信息
	 * 如果有授权返回中文的json格式个人信息
	 * 如果无授权或者失败的几种情况
	 * 	{status: 200, result: res.userInfo, msg: '获取用户信息成功'}
	 * 	{status: 400, result: false, msg: '调取配置信息函数失败'}
	 * 	{status: 400, result: false, msg: '没有授权'}
	 * 	{status: 402, result: false, msg: '获取用户信息函数失败'}
	 * 	{statsu: 403,result: jsonResult.response,msg: '获取用户权限函数调用成功，但是没有拿到信息'} 此种情况应该不会出现但是内部做了一些判断
	 */
checkIsAuthUserInfo() {
  return new Promise((resolve, reject) => {
    my.getSetting({
      success: (res) => {
        console.log('获取用户设置');
        if (res.authSetting['userInfo']) {
          // 已经授权
          my.getOpenUserInfo({
            success(res) {
              // 授权了并且调用成功了
              console.log(res);
              var jsonResult = JSON.parse(res.response);
              if (jsonResult.response.code === '10000') {
                resolve({
                  status: 200,
                  result: jsonResult.response,
                  msg: '获取用户信息成功'
                });
              } else {
                resolve({
                  statsu: 403,
                  result: jsonResult.response,
                  msg: '获取用户权限函数调用成功，但是没有拿到权限'
                })
              }
            },
            fail(err) {
              // 授权了 但是调用方法失败了
              // console.log(err);
              console.log(err);
              resolve({
                status: 402,
                result: false,
                msg: '获取用户信息函数失败'
              });
            }
          });
        } else {
          // 未授权
          resolve({
            status: 400,
            result: false,
            msg: '没有授权'
          });
        }
      },
      fail(err) {
        resolve({
          status: 400,
          result: false,
          msg: '调取配置信息函数失败'
        });
      }
    });
  })
}
```
